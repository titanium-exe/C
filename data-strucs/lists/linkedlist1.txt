#include <stdio.h>
#include <stdlib.h>

// head is a pointer to struc Node. 
// Node* is the pointer

struct Node {
	int data;
	Node* link;
};

Node* head;
head = NULL;

// creating a new node in heap memory 

Node* temp = (Node*)malloc(sizeof(Node));
// dereferencing 
(*temp).data = 2;
(*temp).link = NULL;

head = temp;

temp = (Node*)malloc(sizeof(Node));
temp->data = 5;
temp->link = NULL;

Node* traversor = head;
while(traversor->link != NULL){
	traversor = traversor->link;
}
traversor->link = temp;
